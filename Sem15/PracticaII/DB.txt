DROP DATABASE IF EXISTS CATALOGO;

CREATE DATABASE CATALOGO;

USE CATALOGO;

CREATE TABLE PRODUCTO(
   IDPRO INT NOT NULL auto_increment,
   NOMBRE VARCHAR(80),
   CATEGORIA VARCHAR(80),
   PRECIO  INT,
   PRECIOMAYOR  INT,
   STOCK   INT,
   PRIMARY KEY (IDPRO)
);

USE MYSQL;

GRANT ALL PRIVILEGES ON *.* TO 'catalogo'@'%' IDENTIFIED BY 'admin' WITH GRANT OPTION;
FLUSH PRIVILEGES;

GRANT ALL PRIVILEGES ON *.* TO 'catalogo'@'localhost' IDENTIFIED BY 'admin' WITH GRANT OPTION;
FLUSH PRIVILEGES;

USE CATALOGO;
